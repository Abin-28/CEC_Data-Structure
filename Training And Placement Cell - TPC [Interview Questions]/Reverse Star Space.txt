#include <stdio.h>

int main() {
    int n,j;
    printf("enter limit:");
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
    {for( j=1;j<=i;j++)
    {printf(" ");}
    for(int k=j;k<=n+1;k++)
    {printf("* ");}
    printf("\n");
    }
    
}
