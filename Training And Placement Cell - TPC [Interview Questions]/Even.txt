#include <stdio.h>

int main() {
    int n,c,k=0;
    printf("Enter the number:");
    scanf("%d",&n);
    while(k<=n)
    {   c=0;
        for(int i=0;i<k;i++)
        {
            if(k%2==0)
            {   c=1;
                break;
            }
        }
        if(c==1)
        {
            printf("%d ",k);
        }
        k++;
    }
    

    return 0;
}
